    let match;

    while ((match = linkRegex.exec(html)) !== null) {

    const crawlWebsite = async (url: string, options: CrawlOptions) => {

    const visited = new Set();

    const queue = [{ url, depth: 0 }];

    const assets = new Map();

    while (queue.length > 0) {

    const { url: currentUrl, depth } = queue.shift()!;

    if (visited.has(currentUrl) || depth > options.crawlDepth) {

    continue;

    }

    visited.add(currentUrl);

    const page = await fetchPage(currentUrl);

    const pageAssets = extractAssets(page, options);

    // Add discovered links to queue

    const links = extractLinks(page);

    links.forEach(link => {

    if (shouldCrawl(link, options)) {

    queue.push({ url: link, depth: depth + 1 });

    }

    });

    // Download assets

    for (const asset of pageAssets) {

    if (!assets.has(asset)) {

    const buffer = await downloadAsset(asset);

    assets.set(asset, buffer);

    }

    }

    }

    return generateZip(assets);

    };

    function computeSectionsFor(imgObj) {

    if (!imgObj) return [];

    const xs = [0, ...(mode === 'grid' ? gridXs(imgObj) : vGuides), imgObj.width];

    const ys = [0, ...(mode === 'grid' ? gridYs(imgObj) : hGuides), imgObj.height];

    xs.sort((a,b) => a-b);

    ys.sort((a,b) => a-b);

    const rects = [];

    for(let r = 0; r < ys.length - 1; r++) {

    for(let c = 0; c < xs.length - 1; c++) {

    const x = xs[c], y = ys[r];

    const w = xs[c+1] - x, h = ys[r+1] - y;

    if(w > 0 && h > 0) {

    rects.push({x, y, w, h, idx: rects.length});

    }

    }

    }

    return rects;

    }

    // High-quality section rendering

    function renderSectionBlobFor(imgObj, r, mime='image/png', quality=0.9) {

    return new Promise((resolve) => {

    const off = document.createElement('canvas');

    off.width = r.w;

    off.height = r.h;

    const octx = off.getContext('2d');

    octx.imageSmoothingEnabled = true;

    octx.drawImage(imgObj, r.x, r.y, r.w, r.h, 0, 0, r.w, r.h);

    off.toBlob((blob) => resolve(blob), mime, quality);

    });

    }
