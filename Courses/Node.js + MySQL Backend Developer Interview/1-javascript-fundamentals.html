    1. What is the difference between `==` and `===` in JavaScript?

    Description: `==` performs type coercion before comparison, while `===` compares both value and type without conversion.

    Examples:

    javascript 
    // Loose equality (==) - performs type coercion
5 == "5"        // true (string "5" converted to number)
true == 1       // true (boolean converted to number)
null == undefined // true (special case)
0 == false      // true (both convert to 0)

// Strict equality (===) - no type coercion
5 === "5"       // false (different types)
true === 1      // false (different types)
null === undefined // false (different types)
0 === false     // false (different types)

    2. Explain the concept of hoisting in JavaScript.

    Description: Hoisting moves variable and function declarations to the top of their scope during compilation phase.

    Examples:

    javascript 
    // Variable hoisting with var
console.log(x); // undefined (not ReferenceError)
var x = 5;
// Equivalent to:
// var x;
// console.log(x);
// x = 5;

// Function hoisting
sayHello(); // "Hello!" - works because function is hoisted
function sayHello() {
    console.log("Hello!");
}

// let/const hoisting (Temporal Dead Zone)
console.log(y); // ReferenceError: Cannot access 'y' before initialization
let y = 10;

// Function expression - not hoisted
sayBye(); // TypeError: sayBye is not a function
var sayBye = function() {
    console.log("Bye!");
};

    8. What is prototypal inheritance?

    Description: JavaScript's inheritance mechanism where objects inherit properties and methods from other objects through the prototype chain.

    Examples:

    javascript 
    // Constructor function
function Animal(name) {
    this.name = name;
}

Animal.prototype.speak = function() {
    console.log(`${this.name} makes a sound`);
};

function Dog(name, breed) {
    Animal.call(this, name); // Call parent constructor
    this.breed = breed;
}

// Set up inheritance
Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.constructor = Dog;

Dog.prototype.bark = function() {
    console.log(`${this.name} barks!`);
};

const myDog = new Dog("Buddy", "Golden Retriever");
myDog.speak(); // "Buddy makes a sound" (inherited)
myDog.bark();  // "Buddy barks!" (own method)

// Prototype chain
console.log(myDog.__proto__ === Dog.prototype);           // true
console.log(Dog.prototype.__proto__ === Animal.prototype); // true
console.log(Animal.prototype.__proto__ === Object.prototype); // true

// Object.create example
const parent = {
    greet: function() {
        console.log(`Hello from ${this.name}`);
    }
};

const child = Object.create(parent);
child.name = "Child";
child.greet(); // "Hello from Child"

    10. What is destructuring assignment?

        }
    }

    14. What are getters and setters in JavaScript?

        }
    },
    
    // Computed property
    get info() {
        return `${this.name} is ${this.age} years old`;
    }

    Description: Memoization is an optimization technique that caches function results based on input parameters to avoid redundant calculations.

            return cache[key];

    const cache = new Map();
    
    return function(...args) {
        const key = JSON.stringify(args);
        const now = Date.now();

function deepCopy(obj) {
    // Handle null and non-objects
    if (obj === null || typeof obj !== 'object') {
        return obj;
    }
    
    // Handle Date
    if (obj instanceof Date) {
        return new Date(obj.getTime());
    }
    
    // Handle Array
    if (Array.isArray(obj)) {
        return obj.map(item => deepCopy(item));
    }
    
    // Handle RegExp
    if (obj instanceof RegExp) {
        return new RegExp(obj);
    }
    
    // Handle Objects
    const copiedObj = {};
    for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
            copiedObj[key] = deepCopy(obj[key]);
        }
    }
    
    return copiedObj;
}

// 3. Advanced deep copy with circular reference handling
function advancedDeepCopy(obj, visited = new WeakMap()) {
    // Handle primitives and null
    if (obj === null || typeof obj !== 'object') {
        return obj;
    }
    
    // Handle circular references
    if (visited.has(obj)) {
        return visited.get(obj);
    }
    
    let copy;
    
    // Handle different object types
    if (obj instanceof Date) {
        copy = new Date(obj.getTime());
    } else if (obj instanceof RegExp) {
        copy = new RegExp(obj.source, obj.flags);
    } else if (obj instanceof Map) {
        copy = new Map();
        visited.set(obj, copy);
        for (let [key, value] of obj) {
            copy.set(advancedDeepCopy(key, visited), advancedDeepCopy(value, visited));
        }
        return copy;
    } else if (obj instanceof Set) {
        copy = new Set();
        visited.set(obj, copy);
        for (let value of obj) {
            copy.add(advancedDeepCopy(value, visited));
        }
        return copy;
    } else if (Array.isArray(obj)) {
        copy = [];
        visited.set(obj, copy);
        for (let i = 0; i < obj.length; i++) {
            copy[i] = advancedDeepCopy(obj[i], visited);
        }
        return copy;
    } else {
        copy = {};
        visited.set(obj, copy);
        for (let key in obj) {
            if (obj.hasOwnProperty(key)) {
                copy[key] = advancedDeepCopy(obj[key], visited);
            }
        }
    }
    
    return copy;
}

async function* mergeAsyncGenerators(...generators) {
    const iterators = generators.map(gen => gen[Symbol.asyncIterator]());
    const promises = iterators.map((iter, index) => 
        iter.next().then(result => ({ result, index }))
    );
    
    while (promises.length > 0) {
        const { result, index } = await Promise.race(promises);
        
        if (!result.done) {
            yield result.value;
            
            // Replace the resolved promise with the next one from the same iterator
            promises[index] = iterators[index].next().then(result => ({ result, index }));
        } else {
            // Remove completed iterator
            promises.splice(index, 1);
            iterators.splice(index, 1);
            
            // Update indices for remaining promises
            for (let i = index; i < promises.length; i++) {
                promises[i] = promises[i].then(({ result }) => ({ result, index: i }));
            }
        }
    }
}

class AsyncBuffer {
    constructor(maxSize = 10) {
        this.buffer = [];
        this.maxSize = maxSize;
        this.waitingConsumers = [];
        this.waitingProducers = [];
    }
    
    async push(item) {
        if (this.buffer.length >= this.maxSize) {
            // Wait for space in buffer
            await new Promise(resolve => {
                this.waitingProducers.push(resolve);
            });
        }
        
        this.buffer.push(item);
        
        // Notify waiting consumers
        if (this.waitingConsumers.length > 0) {
            const resolve = this.waitingConsumers.shift();
            resolve();
        }
    }
    
    async pop() {
        if (this.buffer.length === 0) {
            // Wait for items in buffer
            await new Promise(resolve => {
                this.waitingConsumers.push(resolve);
            });
        }
        
        const item = this.buffer.shift();
        
        // Notify waiting producers
        if (this.waitingProducers.length > 0) {
            const resolve = this.waitingProducers.shift();
            resolve();
        }
        
        return item;
    }
    
    async* consume() {
        while (true) {
            yield await this.pop();
        }
    }
}
